@end

    NSInteger sum = 0;

 c. Write a function that prints numbers from 1 to 100 but prints “Hello” if the number is divisible by 3, “World” if the number is divisible by 5, “Hello World” if the number is divisible by both 3 & 5. Use the switch case inside a for loop. 




#import <Foundation/Foundation.h>
@interface PrintMsg : NSObject
- (void) printingMsg;
@end


@implementation PrintMsg
- (void)printingMsg {
    NSInteger i;
    
    for (i = 1; i <= 100; i++) {
       int val = (i % 3 == 0 ? 2 : 0) + (i % 5 == 0 ? 1 : 0);


       switch (val) {
            case 2:
                NSLog(@"Hello");
                break;
                
            case 1:
                NSLog(@"World");
                break;
                
            case 3:
                NSLog(@"Hello World");
                break;
                
            default:
                NSLog(@"%ld", (long)i);
                break;
        }   } }


@end
int main() {
        PrintMsg *s = [[PrintMsg alloc] init];
        [s printingMsg];
    
    return 0;
}




 d. Function that takes two arrays of numbers and returns an array formed by alternatively merging elements from the first array and elements from the second array. e.g: input arrays [10,50,70,42], [78, 90] => return array: [10,78, 50, 90, 70, 42]. Note: you should not modify the input arrays in any way


#import <Foundation/Foundation.h>
@interface CombineArrays : NSObject
- (NSArray *)mergearr:(NSArray *)a1 with:(NSArray *)a2;
@end
@implementation CombineArrays
- (NSArray *)mergearr:(NSArray *)a1 with:(NSArray *)a2 {


   NSMutableArray *res = [[NSMutableArray alloc] init];
   NSInteger i = 0;
    while (i < [a1 count] || i < [a2 count]) {
        if (i < [a1 count]) {
          [res addObject:[a1 objectAtIndex:i]];
        }
        if (i < [a2 count]) {
          [res addObject:[a2 objectAtIndex:i]];
        }
        i++;
    }
    return res;
}
@end
int main() {
        CombineArrays *obj = [[CombineArrays alloc] init];
        NSArray *a1 = [NSArray arrayWithObjects:
                       [NSNumber numberWithInt:10],
                       [NSNumber numberWithInt:50],
                       [NSNumber numberWithInt:70],
                       [NSNumber numberWithInt:42], nil];


        NSArray *a2 = [NSArray arrayWithObjects:
                       [NSNumber numberWithInt:78],
                       [NSNumber numberWithInt:90], nil];
        NSArray *result = [obj mergearr:a1 with:a2];
        NSLog(@"Merged Array : %@", result);
    }
    return 0;
}




 e. Function that takes an array of numbers and returns an array same as the input array but without any duplicates in it. e.g: input array [10,20,30,10,20,30] => return array: [10, 20, 30]. Note: you should not modify the input array in any way


#import <Foundation/Foundation.h>


@interface UniquElement : NSObject
- (NSArray *)Given:(NSArray *)data;
@end


@implementation UniquElement


- (NSArray *)Given:(NSArray *)data {
    NSOrderedSet *set = [NSOrderedSet orderedSetWithArray:data];
    return [set array];
}
@end
int main() {
      UniquElement *s = [[UniquElement alloc] init];
        NSArray *a = [NSArray arrayWithObjects:
                     [NSNumber numberWithInt:10],
                     [NSNumber numberWithInt:20],
                     [NSNumber numberWithInt:30],
                     [NSNumber numberWithInt:10],
                     [NSNumber numberWithInt:20],
                     [NSNumber numberWithInt:30], nil];


        NSArray *res = [s Given:a];
        NSLog(@"Unique Result: %@", res);
    return 0;
}


























 f. Fn that takes an array of characters and shifts the elements in the array in a circular fashion. e.g: input array: [“a”, “b”, “c”, “d”], modify this array to be [“d”,“a”, “b”, “c” ]. Another example. if the input array is now: [“d”,“a”, “b”, “c” ], modify this to be as follows: [“c”, “d”, “a”, “b”]. 


#import <Foundation/Foundation.h>


@interface MoveElement : NSObject
- (void)shiftarray:(NSMutableArray *)p;
@end


@implementation MoveElement


- (void)shiftarray:(NSMutableArray *)p {
    NSInteger n;
    id t;
    n = [p count];
    if (n > 1) {
        t = [p objectAtIndex:n - 1];
        [p removeObjectAtIndex:n - 1];
        [p insertObject:t atIndex:0];
    }  }


@end


int main() {
    
        MoveElement *obj = [[MoveElement alloc] init];
        NSMutableArray *v;
        v = [NSMutableArray arrayWithObjects:@"a", @"b", @"c", @"d", nil];
        [obj shiftarray:v];
        NSLog(@"Res: %@", v);


    return 0;
}

Assume that all keys are strings and all values are numbers. 


#import <Foundation/Foundation.h>


@interface FreqCount : NSObject
- (NSDictionary *)check:(NSString *)s;
@end


@implementation FreqCount


- (NSDictionary *)check:(NSString *)s {
    NSMutableDictionary *d;
    NSInteger i;
    d = [[NSMutableDictionary alloc] init];


    for (i = 0; i < [s length]; i++) {
        NSString *k = [NSString stringWithFormat:@"%C", [s characterAtIndex:i]];
        int v = [[d objectForKey:k] intValue];
        [d setObject:[NSNumber numberWithInt:v + 1] forKey:k];
    }
    return d;
}
@end

        NSString *input = @"abbcccdddd";

        NSLog(@"NSSet : %@", set);

}
Here we can see arr can print all elements but set return only unique characters
